## 如何实现new？
```js
function myNew(Fun, ...args) {
    let obj = {};
    obj.__proto__ = Fun.prototype;
    let res = Fun.apply(obj, args); // 最关键的一步
    return (res instanceof Object ? res : obj);
}
let obj = myNew(One, "XiaoMing", "18");
console.log("newObj:", obj);
```

```js
Function.__proto__ === Function // 虽然是function，但直接相等还是false，注意`Function.prototype`是函数对象，但没有prototype属性
Function.__proto__ === Object.__proto__
Function.prototype.__proto__ === Object.prototype; // true
Object.prototype.__proto__ === null; // true

child.__proto__ === Child.prototype === parent;
parent.__proto__ === Parent.prototype; // true 这个就是继承
Parent.prototype.__proto__ === Object.prototype; // true
```
