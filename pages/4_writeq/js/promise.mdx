## 如何实现promise？
```js
// 简单的，then/catch只有一个回调函数，没有链式调用
// 大致执行流程是func->then->resolve，但有时then在resolve之后
function MyPromise(fn) {
    this.status = 'pending';
    this.res = undefined;
    this.thenCb;
    this.catchCb;
    this.then = (cb) => {
        this.thenCb = cb;
        this.status == 'resolved' && this.thenCb(this.res);
    }
    this.catch = (cb) => {
        this.catchCb = cb;
        this.status == 'rejected' && this.catchCb(this.res);
    }
    const resolve = (res) => {
        this.status = 'resolved';
        this.res = res;
        this.thenCb && this.thenCb(this.res);
    }
    const reject = (res) => {
        this.status = 'rejected';
        this.res = res;
        this.catchCb && this.catchCb(this.res);
    }
    fn(resolve, reject)
}
const promise = new MyPromise((resolve, reject) => resolve(1))
promise.then(res => console.log(res))
```
```js
// 复杂一点的（有链式调用）（但仅有then）
// 简单的，promise已经resolved，这时候一个链式then
// 此then的cb调用在new promise的同步函数里，并且执行new resolve 
// 复杂的，promise还在pending，这时候一个链式then
// 此时push到回调队列中，同样需要执行new resolve，触发下一个promise
// then一个链式调用（串行），一个并行调用（回调数组）
function MyPromise(fn) {
  this.status = 'pending'
  this.res = undefined;
  this.cbs = [];
  this.then = (cb) => {
    return new MyPromise((resolve) => {
      const thenCb = () => {
          const res = cb(this.res);
          resolve(res); 
      };
      this.cbs.push(thenCb);
      if (this.status === 'resolved') {
        thenCb();
      }
    });
  }
  const resolve = (res) => {
    this.res = res;
    this.cbs.forEach((cb) => cb(this.res));
  }
  fn(resolve);
}
const promise = new MyPromise((resolve, reject) => resolve(1))
promise.then(res => console.log(res)).then(res => console.log(res))
```

## 如何实现Promise.all/race？
```js
Promise.myAll = function (promises) {
  let arr = [], count = 0
  return new Promise((resolve, reject) => {
    promises.forEach((item, i) => {
      Promise.resolve(item).then(res => {
        arr[i] = res
        count += 1
        if (count === promises.length) resolve(arr)
      }).catch(reject)
    })
  })
}
Promise.MyRace = function (promises) {
  return new Promise((resolve, reject) => {
    for (const item of promises) {
      Promise.resolve(item).then(resolve).catch(reject)
    }
  })
}
```
## 如何实现async/await？
* 理解async函数需要先理解Generator函数，因为async函数是Generator函数的语法糖。
```js
function co(gen) {
  return new Promise((resolve, reject) => {
    const g = gen();
    function next(param) {
      const { done, value } = g.next(param);
      if (!done) {
        // 未完成 继续递归
        Promise.resolve(value).then((res) => next(res));
      } else {
        // 完成直接重置 Promise 状态
        resolve(value);
      }
    }
    next();
  });
}
```

## 如何实现rxjs的defer函数？
```js
function defer() {
	this.resolve = null;
	this.reject = null;
	this.promise = new Promise((resolve, reject) => {
		this.resolve = resolve;
		this.reject = reject;
	});
}
```
## 如何实现能限流的promise.all？
当我们通过 new Promise 来创建 Promise 的时候，Promise 就会马上执行。
因此需要通过调用函数来创建 Promise，这样就能延迟创建 Promise 了。

```js
// 超简单版
```
